## <a name="example"></a><span data-ttu-id="4740e-107">Exemple</span><span class="sxs-lookup"><span data-stu-id="4740e-107">Example</span></span>

{  
    void G<T>()  
    {  
    }  
  
    public static void Main()  
    {
